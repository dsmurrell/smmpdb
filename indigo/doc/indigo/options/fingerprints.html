Fingerprints options¶

fp-ord-qwords = 25¶

type:integer
default:25
description:Size of “ordinary” part of a fingerprint, in 8-byte blocks.

fp-sim-qwords = 8¶

type:integer
default:8
description:Size of “similarity” part of a fingerprint, in 8-byte blocks.

fp-any-qwords = 15¶

type:integer
default:15
description:Size of “any” part of a fingerprint, in 8-byte blocks.

fp-tau-qwords = 25¶

type:integer
default:25
description:Size of “tautomer” part of a fingerprint, in 8-byte blocks.

fp-ext-enabled = true¶

type:boolean
default:true
description:Sets whether to include or not 3-byte “EXT” part of the fingerprint.
